# Import image
wget http://cdn.download.cirros-cloud.net/0.3.1/cirros-0.3.1-x86_64-disk.img
glance image-create --name="cirros031" --disk-format=qcow2  --container-format=bare --is-public=true < cirros-0.3.1-x86_64-disk.img

# Generate ssh key
ssh-keygen
cd .ssh
nova keypair-add --pub_key id_rsa.pub mykey

# Boot instance
nova boot --flavor 1 --key_name mykey --image cirros031 testvm


# How to Debug Nova-API
1) /nova/nova/cmd/__init__.py
sys.path.append('/opt/eclipse/plugins/org.python.pydev_2.8.2.2013090511/pysrc/')

eventlet.monkey_patch(all=False,socket=True,select=True) 

2)
if you want to debug in one file, add this line to it:
import pydevd;pydevd.settrace('localhost', port=5678, stdoutToServer=True, stderrToServer=True,suspend=True)

As soon as the file has one line above, you can set more breakpoints on it directly

# How to Debug Nova-Compute
1) /opt/stack/nova/nova/service.py
sys.path.append('/opt/eclipse/plugins/org.python.pydev_2.8.2.2013090511/pysrc/')

change "from pydev import pydevd" to "import pydevd"

2) /nova/nova/cmd/__init__.py
eventlet.monkey_patch(all=False,socket=True,select=True) 

3) Start nova-compute with remote debug param
/usr/local/bin/nova-compute --config-file /etc/nova/nova.conf --remote_debug-host 127.0.0.1 --remote_debug-port 5678

4) Set any breakpoints on nova-compute files directly

5) Restart all services
